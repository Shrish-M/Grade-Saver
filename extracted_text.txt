int get_characters (char * file_name) {
    char buffer[50];
    fp = fopen (file_name, "r");
    if (fp == NULL)
        return ERR_ON_OPEN;
    fscanf (fp, " %50c", &buffer[0]);
    for (int i = 49; i >= 0; i--) {
        if (buffer[i] != '0')
            printf ("%c", buffer[i]);
    }
}
